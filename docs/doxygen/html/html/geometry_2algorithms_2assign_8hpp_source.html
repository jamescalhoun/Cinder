144 {

146 
147  bool const same_point_order =
148  point_order<Geometry1>::value == point_order<Geometry2>::value;
149  bool const same_closure =
150  closure<Geometry1>::value == closure<Geometry2>::value;
151 
152  BOOST_MPL_ASSERT_MSG

155  , (types<Geometry1, Geometry2>)
156  );
157  BOOST_MPL_ASSERT_MSG
