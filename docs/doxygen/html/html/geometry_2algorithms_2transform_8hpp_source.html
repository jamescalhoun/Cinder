55 struct transform_box
56 {
57  template <typename Box1, typename Box2, typename Strategy>
58  static inline bool apply(Box1 const& b1, Box2& b2,
59  Strategy const& strategy)
60  {
61  typedef typename point_type<Box1>::type point_type1;
62  typedef typename point_type<Box2>::type point_type2;
63 
64  point_type1 lower_left, upper_right;
65  detail::assign::assign_box_2d_corner<min_corner, min_corner>(
66  b1, lower_left);
67  detail::assign::assign_box_2d_corner<max_corner, max_corner>(
68  b1, upper_right);
69 
70  point_type2 p1, p2;
71  if (strategy.apply(lower_left, p1) && strategy.apply(upper_right, p2))
72  {
73  // Create a valid box and therefore swap if necessary
74  typedef typename coordinate_type<point_type2>::type coordinate_type;
75  coordinate_type x1 = geometry::get<0>(p1)
76  , y1 = geometry::get<1>(p1)
77  , x2 = geometry::get<0>(p2)
78  , y2 = geometry::get<1>(p2);
79 
80  if (x1 > x2) { std::swap(x1, x2); }
81  if (y1 > y2) { std::swap(y1, y2); }
82 
83  set<min_corner, 0>(b2, x1);
84  set<min_corner, 1>(b2, y1);
85  set<max_corner, 0>(b2, x2);
86  set<max_corner, 1>(b2, y2);
87 
88  return true;
89  }
90  return false;
91  }
92 };

94 struct transform_box_or_segment
95 {
96  template <typename Geometry1, typename Geometry2, typename Strategy>
97  static inline bool apply(Geometry1 const& source, Geometry2& target,
98  Strategy const& strategy)
99  {
100  typedef typename point_type<Geometry1>::type point_type1;
101  typedef typename point_type<Geometry2>::type point_type2;
102 
103  point_type1 source_point[2];
104  geometry::detail::assign_point_from_index<0>(source, source_point[0]);
105  geometry::detail::assign_point_from_index<1>(source, source_point[1]);
106 
107  point_type2 target_point[2];
108  if (strategy.apply(source_point[0], target_point[0])
109  && strategy.apply(source_point[1], target_point[1]))
110  {
111  geometry::detail::assign_point_to_index<0>(target_point[0], target);
112  geometry::detail::assign_point_to_index<1>(target_point[1], target);
113  return true;
114  }
115  return false;
116  }
117 };

122  typename PointOut,
123  typename OutputIterator,
124  typename Range,
125  typename Strategy
126 >
127 inline bool transform_range_out(Range const& range,
128  OutputIterator out, Strategy const& strategy)
129 {
130  PointOut point_out;
131  for(typename boost::range_iterator<Range const>::type
132  it = boost::begin(range);
133  it != boost::end(range);
134  ++it)
135  {
136  if (! transform_point::apply(*it, point_out, strategy))
137  {
138  return false;
139  }
140  *out++ = point_out;
141  }

146 struct transform_polygon
147 {
148  template <typename Polygon1, typename Polygon2, typename Strategy>
149  static inline bool apply(Polygon1 const& poly1, Polygon2& poly2,
150  Strategy const& strategy)
151  {
152  typedef typename ring_type<Polygon1>::type ring1_type;
153  typedef typename ring_type<Polygon2>::type ring2_type;
154  typedef typename point_type<Polygon2>::type point2_type;
155 
156  geometry::clear(poly2);
157 
158  if (!transform_range_out<point2_type>(exterior_ring(poly1),
159  std::back_inserter(exterior_ring(poly2)), strategy))
160  {
161  return false;
162  }
163 
164  // Note: here a resizeable container is assumed.
165  traits::resize
166  <
167  typename boost::remove_reference
168  <
169  typename traits::interior_mutable_type<Polygon2>::type
170  >::type
171  >::apply(interior_rings(poly2), num_interior_rings(poly1));
172 
173  typename interior_return_type<Polygon1 const>::type rings1
174  = interior_rings(poly1);
175  typename interior_return_type<Polygon2>::type rings2
176  = interior_rings(poly2);
177  BOOST_AUTO_TPL(it1, boost::begin(rings1));
178  BOOST_AUTO_TPL(it2, boost::begin(rings2));

186  }
