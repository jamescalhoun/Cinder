62  template <typename T, typename OutputIterator>
63  OutputIterator encode_one ( T val, OutputIterator out ) {
64  const std::size_t num_hex_digits = 2 * sizeof ( T );
65  char res [ num_hex_digits ];
66  char *p = res + num_hex_digits;
67  for ( std::size_t i = 0; i < num_hex_digits; ++i, val >>= 4 )
68  *--p = "0123456789ABCDEF" [ val & 0x0F ];
69  return std::copy ( res, res + num_hex_digits, out );
70  }

76  unsigned hex_char_to_int ( char c ) {
77  if ( c >= '0' && c <= '9' ) return c - '0';
78  if ( c >= 'A' && c <= 'F' ) return c - 'A' + 10;
79  if ( c >= 'a' && c <= 'f' ) return c - 'a' + 10;
80  BOOST_THROW_EXCEPTION (non_hex_input() << bad_char (c));
81  return 0; // keep dumb compilers happy
82  }
83  }

123 // What can we assume here about the inputs?
124 // is std::iterator_traits<InputIterator>::value_type always 'char' ?
125 // Could it be wchar_t, say? Does it matter?
126 // We are assuming ASCII for the values - but what about the storage?
127  template <typename InputIterator, typename OutputIterator, typename EndPred>
128  typename boost::enable_if<boost::is_integral<typename hex_iterator_traits<OutputIterator>::value_type>, OutputIterator>::type
129  decode_one ( InputIterator &first, InputIterator last, OutputIterator out, EndPred pred ) {
130  typedef typename hex_iterator_traits<OutputIterator>::value_type T;
131  T res (0);
132 
133  // Need to make sure that we get can read that many chars here.
134  for ( std::size_t i = 0; i < 2 * sizeof ( T ); ++i, ++first ) {
135  if ( pred ( first, last ))
136  BOOST_THROW_EXCEPTION (not_enough_input ());
137  res = ( 16 * res ) + hex_char_to_int (static_cast<char> (*first));
138  }
139 
140  *out = res;
141  return ++out;
142  }
144  }
