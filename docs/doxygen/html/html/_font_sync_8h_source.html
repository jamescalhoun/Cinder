60 /* Matching Options */
61 typedef UInt32 FNSMatchOptions;
62 enum {
63  kFNSMatchNames = 0x00000001, /* font names must match */
64  kFNSMatchTechnology = 0x00000002, /* scaler technology must match */
65  kFNSMatchGlyphs = 0x00000004, /* glyph data must match */
66  kFNSMatchEncodings = 0x00000008, /* cmaps must match */
67  kFNSMatchQDMetrics = 0x00000010, /* QuickDraw Text metrics must match */
68  kFNSMatchATSUMetrics = 0x00000020, /* ATSUI metrics (incl. vertical) must match */
69  kFNSMatchKerning = 0x00000040, /* kerning data must match */
70  kFNSMatchWSLayout = 0x00000080, /* WorldScript layout tables must match */
71  kFNSMatchAATLayout = 0x00000100, /* AAT (incl. OpenType) layout tables must match */
72  kFNSMatchPrintEncoding = 0x00000200, /* PostScript font and glyph names and re-encoding vector must match */
73  kFNSMissingDataNoMatch = (unsigned long)0x80000000, /* treat missing data as mismatch */
74  kFNSMatchAll = (unsigned long)0xFFFFFFFF, /* everything must match */
75  kFNSMatchDefaults = 0 /* use global default match options */
76 };
