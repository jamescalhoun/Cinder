23  thread_clock::time_point thread_clock::now( ) BOOST_NOEXCEPT
24  {
25  struct timespec ts;
26 #if defined CLOCK_THREAD_CPUTIME_ID
27  // get the timespec associated to the thread clock
28  if ( ::clock_gettime( CLOCK_THREAD_CPUTIME_ID, &ts ) )
29 #else
30  // get the current thread
31  pthread_t pth=pthread_self();
32  // get the clock_id associated to the current thread
33  clockid_t clock_id;
34  pthread_getcpuclockid(pth, &clock_id);
35  // get the timespec associated to the thread clock
36  if ( ::clock_gettime( clock_id, &ts ) )
37 #endif

42  // transform to nanoseconds
