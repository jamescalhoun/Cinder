232 /*
233  IDHGenericEvent
234  An IDH will often have to post events from at interrupt time. Since memory
235  allocation cannot occur from the interrupt handler, the IDH can preallocate
236  storage needed for handling the event by creating some IDHGenericEvent items.
237  Subsequently, when an event is generated, the type of event (specified in the
238  IDHEventHeader) will dictate how the IDHGenericEvent should be interpretted.
239 
240  IMPORTANT NOTE : This means that a specific event structure can NEVER be greater
241  than the size of the generic one.
242 
243 */
244 struct IDHGenericEvent {
245  IDHEventHeader eventHeader;
246  UInt32 pad[4];
247 };
248 typedef struct IDHGenericEvent IDHGenericEvent;
