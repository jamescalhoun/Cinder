365 /*
366  * TQAVGouraud is used for Gouraud shading. Each vertex specifies position, color and Z.
367  *
368  * Alpha is always treated as indicating transparency. Drawing engines which don't
369  * support Z-sorted rendering use the back-to-front transparency blending functions
370  * shown below. (ARGBsrc are the source (new) values, ARGBdest are the destination
371  * (previous) pixel values.)
372  *
373  * Premultiplied Interpolated
374  *
375  * A = 1 - (1 - Asrc) * (1 - Adest) A = 1 - (1 - Asrc) * (1 - Adest)
376  * R = (1 - Asrc) * Rdest + Rsrc R = (1 - Asrc) * Rdest + Asrc * Rsrc
377  * G = (1 - Asrc) * Gdest + Gsrc G = (1 - Asrc) * Gdest + Asrc * Gsrc
378  * B = (1 - Asrc) * Bdest + Bsrc B = (1 - Asrc) * Bdest + Asrc * Bsrc
379  *
380  * Note that the use of other blending modes to implement antialiasing is performed
381  * automatically by the drawing engine when the kQATag_Antialias variable !=
382  * kQAAntiAlias_Fast. The driving software should continue to use the alpha fields
383  * for transparency even when antialiasing is being used (the drawing engine will
384  * resolve the multiple blending requirements as best as it can).
385  *
386  * Drawing engines which perform front-to-back Z-sorted rendering should replace
387  * the blending function shown above with the equivalent front-to-back formula.
388  */
389 struct TQAVGouraud {
390  float x; /* X pixel coordinate, 0.0 <= x < width */
391  float y; /* Y pixel coordinate, 0.0 <= y < height */
392  float z; /* Z coordinate, 0.0 <= z <= 1.0 */
393  float invW; /* 1 / w; required only when kQAPerspectiveZ_On is set */
394 
395  float r; /* Red, 0.0 <= r <= 1.0 */
396  float g; /* Green, 0.0 <= g <= 1.0 */
397  float b; /* Blue, 0.0 <= b <= 1.0 */
398  float a; /* Alpha, 0.0 <= a <= 1.0, 1.0 is opaque */
399 };
400 typedef struct TQAVGouraud TQAVGouraud;
401 /*
402  * TQAVTexture is used for texture mapping. The texture mapping operation
403  * is controlled by the kQATag_TextureOp variable, which is a mask of
404  * kQATextureOp_None/Modulate/Highlight/Decal. Below is pseudo-code for the
405  * texture shading operation:
406  *
407  * texPix = TextureLookup (uq/q, vq/q);
408  * if (kQATextureOp_Decal)
409  * {
410  * texPix.r = texPix.a * texPix.r + (1 - texPix.a) * r;
411  * texPix.g = texPix.a * texPix.g + (1 - texPix.a) * g;
412  * texPix.b = texPix.a * texPix.b + (1 - texPix.a) * b;
413  * texPix.a = a;
414  * }
415  * else
416  * {
417  * texPix.a = texPix.a * a;
418  * }
419  * if (kQATextureOp_Modulate)
420  * {
421  * texPix.r *= kd_r; // Clamped to prevent overflow
422  * texPix.g *= kd_g; // Clamped to prevent overflow
423  * texPix.b *= kd_b; // Clamped to prevent overflow
424  * }
425  * if (kQATextureOp_Highlight)
426  * {
427  * texPix.r += ks_r; // Clamped to prevent overflow
428  * texPix.g += ks_g; // Clamped to prevent overflow
429  * texPix.b += ks_b; // Clamped to prevent overflow
430  * }
431  *
432  * After computation of texPix, transparency blending (as shown
433  * above for TQAVGouraud) is performed.
434  */

1047 /************************************************************************************************
1048  *
1049  * Macro definition for easily setting the texture priority bits in the flags field of
1050  * QATextureNew(), and QABitMapNew().
1051  *
1052  * The priority value must be a number between 0 and 15. 0 is top priority, and 15 is lowest
1053  * priority. Note that this is different than OpenGL which uses floats from 0.0 to 1.0 where
1054  * 1.0 is top priority. In RAVE, this system is more like texture ranking, thus 0 is top rank,
1055  * and 15 is the last ranking.
1056  *
1057  * Texture prioritization is mainly used for systems which support AGP memory. This allows the
1058  * hardware driver to determine which textures to put into faster VRAM and which to put into
1059  * slower AGP RAM. If you dont care about texture priority, then you dont need to worry about
1060  * this macro or the priority bits since the default will always be top priority when all 4 bits
1061  * of this flag field are 0's.
1062  *
1063  * To upload a texture with high priority, do the following:
1064  *
1065  * unsigned long flags = kQATexture_Mipmap; // Initialize flags
1066  *
1067  * flags |= QACalculatePriorityBits(0); // Setting top priority (this actually does
1068  * // nothing since the bits are still 0000)
1069  * QATextureNew(myEngine, flags, pixelType, image, &raveTexture);
1070  *
1071  *
1072  * To do a texture with a middle priority of 8, do the following:
1073  *
1074  * unsigned long flags = kQATexture_HighCompression; // Initialize flags
1075  *
1076  * flags |= QACalculatePriorityBits(8); // Setting middle priority
1077  *
1078  * QATextureNew(myEngine, flags, pixelType, image, &raveTexture);
1079  *
1080  ***********************************************************************************************/
1081
