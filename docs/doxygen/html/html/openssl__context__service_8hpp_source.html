297  // Use the specified file to obtain the temporary Diffie-Hellman parameters.
298  boost::system::error_code use_tmp_dh_file(impl_type& impl,
299  const std::string& filename, boost::system::error_code& ec)
300  {
301  ::BIO* bio = ::BIO_new_file(filename.c_str(), "r");
302  if (!bio)
303  {
304  ec = boost::asio::error::invalid_argument;
305  return ec;
306  }
307 
308  ::DH* dh = ::PEM_read_bio_DHparams(bio, 0, 0, 0);
309  if (!dh)
310  {
311  ::BIO_free(bio);
312  ec = boost::asio::error::invalid_argument;
313  return ec;
314  }
315 
316  ::BIO_free(bio);
317  int result = ::SSL_CTX_set_tmp_dh(impl, dh);
318  ::DH_free(dh);
319  if (result != 1)
320  {
321  ec = boost::asio::error::invalid_argument;
322  return ec;
323  }
324 
325  ec = boost::system::error_code();
326  return ec;
327  }

329  static int password_callback(char* buf, int size, int purpose, void* data)
330  {
331  using namespace std; // For strncat and strlen.
332 
333  if (data)
334  {
335  password_callback_type* callback =
336  static_cast<password_callback_type*>(data);
337  std::string passwd = (*callback)(static_cast<std::size_t>(size),
338  purpose ? context_base::for_writing : context_base::for_reading);
339  *buf = '\0';
340  strncat(buf, passwd.c_str(), size);
341  return strlen(buf);
342  }
343 
344  return 0;
345  }

347  // Set the password callback.
348  template <typename Password_Callback>
349  boost::system::error_code set_password_callback(impl_type& impl,
350  Password_Callback callback, boost::system::error_code& ec)
351  {
352  // Allocate callback function object if not already present.
353  if (impl->default_passwd_callback_userdata)
354  {
355  password_callback_type* callback_function =
356  static_cast<password_callback_type*>(
357  impl->default_passwd_callback_userdata);
358  *callback_function = callback;
359  }
360  else
361  {
362  password_callback_type* callback_function =
363  new password_callback_type(callback);
364  impl->default_passwd_callback_userdata = callback_function;
365  }
366 
367  // Set the password callback.
368  SSL_CTX_set_default_passwd_cb(impl,
369  &openssl_context_service::password_callback);
370 
371  ec = boost::system::error_code();
372  return ec;
373  }
