145  std::size_t fill()
146  {
147  detail::buffer_resize_guard<detail::buffered_stream_storage>
148  resize_guard(storage_);
149  std::size_t previous_size = storage_.size();
150  storage_.resize(storage_.capacity());
151  storage_.resize(previous_size + next_layer_.read_some(buffer(
152  storage_.data() + previous_size,
153  storage_.size() - previous_size)));
154  resize_guard.commit();

287  template <typename MutableBufferSequence, typename ReadHandler>
288  void async_read_some(const MutableBufferSequence& buffers,
289  ReadHandler handler)
290  {
291  if (boost::asio::buffer_size(buffers) == 0)
292  {
293  get_io_service().post(detail::bind_handler(
294  handler, boost::system::error_code(), 0));
295  }
296  else if (storage_.empty())
297  {
298  async_fill(read_some_handler<MutableBufferSequence, ReadHandler>(
299  get_io_service(), storage_, buffers, handler));
300  }
301  else
302  {
303  std::size_t length = copy(buffers);
304  get_io_service().post(detail::bind_handler(
305  handler, boost::system::error_code(), length));
306  }
307  }
