98 /* You can specify routing information for a file by including a 'rout' 0
99  resource in it and setting the kExtendedFlagHasRoutingInfo bit in the extended
100  Finder flags.
101  The 'rout' resource is an array of RoutingResourceEntry. Each entry is considered
102  in turn. The first matching entry is used.
103  If the creator and fileType match the file being dropped and targetFolder match
104  the folder ID of the folder being dropped onto, then the file is rerouted
105  into the specified destination folder.
106  The only target folder currently supported is the system folder,
107  kSystemFolderType = 'macs'.
108 */
109 enum {
110  kRoutingResourceType = FOUR_CHAR_CODE('rout'),
111  kRoutingResourceID = 0
112 };
113

169  kIsShared = 0x0040, /* Files only (Applications only) */

176  /* Clear if the file contains desktop database */
177  /* resources ('BNDL', 'FREF', 'open', 'kind'...) */

181  kHasCustomIcon = 0x0400, /* Files and folders */
