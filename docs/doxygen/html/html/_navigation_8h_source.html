158  * A NavDialogRef is an opaque reference to an instance of a

160  * any of the NavCreate*Dialog functions and is later disposed with

176 /*
177  * NavUserAction
178  *
179  * Summary:
180  * Indicates an action taken by the user
181  *
182  * Discussion:
183  * When the user clicks a button at the bottom of a Nav Services
184  * dialog (or makes an equivalent mouse or key gesture), a
185  * kNavCBUserAction event is sent to the client's event proc
186  * indicating which action was taken. Often, the action also
187  * dismisses the dialog. User action events are only generated when
188  * using dialogs created from a NavCreate*Dialog function. In the
189  * special case of a modeless GetFile dialog (supported only on Mac
190  * OS X), the user can option-click on the open button to keep the
191  * dialog from being dismissed, but the kNavCBUserAction event is
192  * sent so the client can get the reply record and open the selected
193  * files.
194  */

774 /*
775  * NavReplyRecord
776  *
777  * Summary:
778  * A structure describing the results of a Nav Services dialog
779  *
780  * Discussion:
781  * A reply record is the result of a Nav Services file dialog. Using
782  * the older API, which is always modal, the client passes the
783  * address of a reply record when invoking the dialog. In the Carbon
784  * API, dialogs may also be window modal or modeless, so the client
785  * requests the reply record by calling NavDialogGetReply when a
786  * kNavCBUserAction event is received. Either way, a reply record
787  * should be disposed of using NavDisposeReply.
788  */
789 struct NavReplyRecord {
790 
791  /*
792  * The version of the structure. The first public version of the
793  * structure was version 0. Fields added after version 0, starting
794  * with the saveFileName field, are noted below.
795  */
796  UInt16 version;
797 
798  /*
799  * True if the reply contains a non-null selection
800  */
801  Boolean validRecord;
802 
803  /*
804  * True if this reply is from a PutFile dialog and the file to be
805  * saved already exists and needs to be replaced. The user has
806  * already been warned unless the kNavDontConfirmReplacement option
807  * flag is used.
808  */
809  Boolean replacing;
810 
811  /*
812  * True if this reply is from a PutFile dialog and the user wants to
813  * save the file as stationery.
814  */
815  Boolean isStationery;
816 
817  /*
818  * True if translation was performed on the file(s) to be opened or
819  * if transtlation will be needed on the file to be saved.
820  */
821  Boolean translationNeeded;
822 
823  /*
824  * For GetFile or Choose dialogs, a list of items chosen by the user.
825  * For the older NavPutFile dialog, a list containing one item: an
826  * FSSpec of the file to be saved. ** IMPORTANT NOTE *** For the new
827  * Carbon-only PutFile dialog created with NavCreatePutFileDialog,
828  * the selection is a list containing one item: the DIRECTORY where
829  * the file is to be saved. The file name is obtained from the
830  * saveFileName field. When using the original modal API, each
831  * descriptor will contain an FSSpec (typeFSS). When using the new
832  * Carbon-only dialogs created via the NavCreate*Dialog functions,
833  * each descriptor could contain either an FSSpec (typeFSS, used on
834  * Mac OS 8 or 9) or an FSRef (typeFSRef, used on Mac OS X). This
835  * divergence is caused by the need to use FSRef (for Unicode/HFS+
836  * support) on Mac OS X, while being unable to provide FSRefs on Mac
837  * OS 8.6.
838  */
839  AEDescList selection;
840 
841  /*
842  * For NavPutFile: the script system associated with the name of the
843  * file to be saved.
844  */
845  ScriptCode keyScript;
846 
847  /*

850  * translation that was performed (GetFile) or needs to be performed
851  * (PutFile) on that item.
852  */

857  */
858  UInt32 reserved1;
859

862  * field contains the true file name to saved, even if the extension
863  * will be hidden from the user. This field was added in structure
864  * version 1.
865  */
866  CFStringRef saveFileName;
867
