129  static buffers_iterator begin(const BufferSequence& buffers)
130 #if BOOST_WORKAROUND(__GNUC__, == 4) && BOOST_WORKAROUND(__GNUC_MINOR__, == 3)
131  __attribute__ ((__noinline__))
132 #endif
133  {
134  buffers_iterator new_iter;
135  new_iter.begin_ = buffers.begin();
136  new_iter.current_ = buffers.begin();
137  new_iter.end_ = buffers.end();
138  while (new_iter.current_ != new_iter.end_)
139  {
140  new_iter.current_buffer_ = *new_iter.current_;
141  if (boost::asio::buffer_size(new_iter.current_buffer_) > 0)
142  break;
143  ++new_iter.current_;
144  }

314  // Increment the iterator.
315  void increment()
316  {
317  BOOST_ASSERT(current_ != end_ && "iterator out of bounds");
318  ++position_;
319 
320  // Check if the increment can be satisfied by the current buffer.
321  ++current_buffer_position_;
322  if (current_buffer_position_ != boost::asio::buffer_size(current_buffer_))
323  return;
324 
325  // Find the next non-empty buffer.
326  ++current_;
327  current_buffer_position_ = 0;
328  while (current_ != end_)
329  {
330  current_buffer_ = *current_;
331  if (boost::asio::buffer_size(current_buffer_) > 0)
332  return;
333  ++current_;
334  }
335  }

337  // Decrement the iterator.
338  void decrement()
339  {
340  BOOST_ASSERT(position_ > 0 && "iterator out of bounds");
341  --position_;
342 
343  // Check if the decrement can be satisfied by the current buffer.
344  if (current_buffer_position_ != 0)
345  {
346  --current_buffer_position_;
347  return;
348  }
349 
350  // Find the previous non-empty buffer.
351  typename BufferSequence::const_iterator iter = current_;
352  while (iter != begin_)
353  {
354  --iter;
355  buffer_type buffer = *iter;
356  std::size_t buffer_size = boost::asio::buffer_size(buffer);
357  if (buffer_size > 0)
358  {
359  current_ = iter;
360  current_buffer_ = buffer;
361  current_buffer_position_ = buffer_size - 1;
362  return;
363  }
364  }
365  }

367  // Advance the iterator by the specified distance.
368  void advance(std::ptrdiff_t n)
369  {
370  if (n > 0)
371  {
372  BOOST_ASSERT(current_ != end_ && "iterator out of bounds");
373  for (;;)
374  {
375  std::ptrdiff_t current_buffer_balance
376  = boost::asio::buffer_size(current_buffer_)
377  - current_buffer_position_;
378 
379  // Check if the advance can be satisfied by the current buffer.
380  if (current_buffer_balance > n)
381  {
382  position_ += n;
383  current_buffer_position_ += n;
384  return;
385  }
386 
387  // Update position.
388  n -= current_buffer_balance;
389  position_ += current_buffer_balance;
390 
391  // Move to next buffer. If it is empty then it will be skipped on the
392  // next iteration of this loop.
393  if (++current_ == end_)
394  {
395  BOOST_ASSERT(n == 0 && "iterator out of bounds");
396  current_buffer_ = buffer_type();
397  current_buffer_position_ = 0;
398  return;
399  }
400  current_buffer_ = *current_;
401  current_buffer_position_ = 0;
402  }
403  }
404  else if (n < 0)
405  {
406  std::size_t abs_n = -n;
407  BOOST_ASSERT(position_ >= abs_n && "iterator out of bounds");
408  for (;;)
409  {
410  // Check if the advance can be satisfied by the current buffer.
411  if (current_buffer_position_ >= abs_n)
412  {
413  position_ -= abs_n;
414  current_buffer_position_ -= abs_n;
415  return;
416  }
417 
418  // Update position.
419  abs_n -= current_buffer_position_;
420  position_ -= current_buffer_position_;
421 
422  // Check if we've reached the beginning of the buffers.
423  if (current_ == begin_)
424  {
425  BOOST_ASSERT(abs_n == 0 && "iterator out of bounds");
426  current_buffer_position_ = 0;
427  return;
428  }
429 
430  // Find the previous non-empty buffer.
431  typename BufferSequence::const_iterator iter = current_;
432  while (iter != begin_)
433  {
434  --iter;
435  buffer_type buffer = *iter;
436  std::size_t buffer_size = boost::asio::buffer_size(buffer);
437  if (buffer_size > 0)
438  {
439  current_ = iter;
440  current_buffer_ = buffer;
441  current_buffer_position_ = buffer_size;
442  break;
443  }
444  }
445  }
446  }
447  }
