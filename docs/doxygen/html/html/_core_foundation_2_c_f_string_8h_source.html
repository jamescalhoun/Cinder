20 /*
21 Please note: CFStrings are conceptually an array of Unicode characters.
22 However, in general, how a CFString stores this array is an implementation
23 detail. For instance, CFString might choose to use an array of 8-bit characters;
24 to store its contents; or it might use multiple blocks of memory; or whatever.
25 Furthermore, the implementation might change depending on the default
26 system encoding, the user's language, the OS, or even a given release.
27 
28 What this means is that you should use the following advanced functions with care:
29 
30  CFStringGetPascalStringPtr()
31  CFStringGetCStringPtr()
32  CFStringGetCharactersPtr()
33 
34 These functions are provided for optimization only. They will either return the desired
35 pointer quickly, in constant time, or they return NULL. They might choose to return NULL
36 for many reasons; for instance it's possible that for users running in different
37 languages these sometimes return NULL; or in a future OS release the first two might
38 switch to always returning NULL. Never observing NULL returns in your usages of these
39 functions does not mean they won't ever return NULL. (But note the CFStringGetCharactersPtr()
40 exception mentioned further below.)
41 
42 In your usages of these functions, if you get a NULL return, use the non-Ptr version
43 of the functions as shown in this example:
44 
45  Str255 buffer;
46  StringPtr ptr = CFStringGetPascalStringPtr(str, encoding);
47  if (ptr == NULL) {
48  if (CFStringGetPascalString(str, buffer, 256, encoding)) ptr = buffer;
49  }
50 
51 Note that CFStringGetPascalString() or CFStringGetCString() calls might still fail --- but
52 that will happen in two circumstances only: The conversion from the UniChar contents of CFString
53 to the specified encoding fails, or the buffer is too small. If they fail, that means
54 the conversion was not possible.
55 
56 If you need a copy of the buffer in the above example, you might consider simply
57 calling CFStringGetPascalString() in all cases --- CFStringGetPascalStringPtr()
58 is simply an optimization.
59 
60 In addition, the following functions, which create immutable CFStrings from developer
61 supplied buffers without copying the buffers, might have to actually copy
62 under certain circumstances (If they do copy, the buffer will be dealt with by the
63 "contentsDeallocator" argument.):
64 
65  CFStringCreateWithPascalStringNoCopy()
66  CFStringCreateWithCStringNoCopy()
67  CFStringCreateWithCharactersNoCopy()
68 
69 You should of course never depend on the backing store of these CFStrings being
70 what you provided, and in other no circumstance should you change the contents
71 of that buffer (given that would break the invariant about the CFString being immutable).
72 
73 Having said all this, there are actually ways to create a CFString where the backing store
74 is external, and can be manipulated by the developer or CFString itself:
75 
76  CFStringCreateMutableWithExternalCharactersNoCopy()
77  CFStringSetExternalCharactersNoCopy()
78 
79 A "contentsAllocator" is used to realloc or free the backing store by CFString.
80 kCFAllocatorNull can be provided to assure CFString will never realloc or free the buffer.
81 Developer can call CFStringSetExternalCharactersNoCopy() to update
82 CFString's idea of what's going on, if the buffer is changed externally. In these
83 strings, CFStringGetCharactersPtr() is guaranteed to return the external buffer.
84 
85 These functions are here to allow wrapping a buffer of UniChar characters in a CFString,
86 allowing the buffer to passed into CFString functions and also manipulated via CFString
87 mutation functions. In general, developers should not use this technique for all strings,
88 as it prevents CFString from using certain optimizations.
89 */

114 /* Macro to allow creation of compile-time constant strings; the argument should be a constant string.
115 
116 CFSTR(), not being a "Copy" or "Create" function, does not return a new
117 reference for you. So, you should not release the return value. This is
118 much like constant C or Pascal strings --- when you use "hello world"
119 in a program, you do not free it.
120 
121 However, strings returned from CFSTR() can be retained and released in a
122 properly nested fashion, just like any other CF type. That is, if you pass
123 a CFSTR() return value to a function such as SetMenuItemWithCFString(), the
124 function can retain it, then later, when it's done with it, it can release it.
125 
126 At this point non-7 bit characters (that is, characters > 127) in CFSTR() are not
127 supported and using them will lead to unpredictable results. This includes escaped
128 (\nnn) characters whose values are > 127. Even if it works for you in testing,
129 it might not work for a user with a different language preference.
130 */
131 #ifdef __CONSTANT_CFSTRINGS__
132 #define CFSTR(cStr) ((CFStringRef) __builtin___CFStringMakeConstantString ("" cStr ""))
133 #else
134 #define CFSTR(cStr) __CFStringMakeConstantString("" cStr "")
135 #endif

617 /* The next two functions allow fast access to the contents of a string,
618  assuming you are doing sequential or localized accesses. To use, call
619  CFStringInitInlineBuffer() with a CFStringInlineBuffer (on the stack, say),
620  and a range in the string to look at. Then call CFStringGetCharacterFromInlineBuffer()
621  as many times as you want, with a index into that range (relative to the start
622  of that range). These are INLINE functions and will end up calling CFString only
623  once in a while, to fill a buffer. CFStringGetCharacterFromInlineBuffer() returns 0 if
624  a location outside the original range is specified.
625 */
626 #define __kCFStringInlineBufferLength 64
627 typedef struct {
628  UniChar buffer[__kCFStringInlineBufferLength];
629  CFStringRef theString;
630  const UniChar *directBuffer;
631  CFRange rangeToBuffer; /* Range in string to buffer */
632  CFIndex bufferedRangeStart; /* Start of range currently buffered (relative to rangeToBuffer.location) */
633  CFIndex bufferedRangeEnd; /* bufferedRangeStart + number of chars actually buffered */
634 } CFStringInlineBuffer;

638  buf->theString = str;
639  buf->rangeToBuffer = range;

644 CF_INLINE UniChar CFStringGetCharacterFromInlineBuffer(CFStringInlineBuffer *buf, CFIndex idx) {
645  if (buf->directBuffer) {
646  if (idx < 0 || idx >= buf->rangeToBuffer.length) return 0;
647  return buf->directBuffer[idx + buf->rangeToBuffer.location];
648  }
649  if (idx >= buf->bufferedRangeEnd || idx < buf->bufferedRangeStart) {
650  if (idx < 0 || idx >= buf->rangeToBuffer.length) return 0;
651  if ((buf->bufferedRangeStart = idx - 4) < 0) buf->bufferedRangeStart = 0;
652  buf->bufferedRangeEnd = buf->bufferedRangeStart + __kCFStringInlineBufferLength;
653  if (buf->bufferedRangeEnd > buf->rangeToBuffer.length) buf->bufferedRangeEnd = buf->rangeToBuffer.length;
654  CFStringGetCharacters(buf->theString, CFRangeMake(buf->rangeToBuffer.location + buf->bufferedRangeStart, buf->bufferedRangeEnd - buf->bufferedRangeStart), buf->buffer);
655  }
656  return buf->buffer[idx - buf->bufferedRangeStart];
657 }
