159  : is_sorted(false)

166  tail_impl(tail_impl const &that)
167  : is_sorted(that.is_sorted)
168  , indices(that.indices)
169  , samples(that.samples)
170  {
171  this->indices.reserve(this->samples.size());
172  }
173 
174  // This just stores the heap and the samples.
175  // In operator()() below, if we are adding a new sample
176  // to the sample cache, we force all the
177  // tail_variates to update also. (It's not
178  // good enough to wait for the accumulator_set to do it
179  // for us because then information about whether a sample
180  // was stored and where is lost, and would need to be
181  // queried at runtime, which would be slow.) This is
182  // implemented as a filtered visitation over the stats,
183  // which we can access because args[accumulator] gives us
184  // all the stats.
185 
186  template<typename Args>
187  void operator ()(Args const &args)
188  {
189  if(this->indices.size() < this->samples.size())
190  {
191  this->indices.push_back(this->indices.size());
192  this->assign(args, this->indices.back());
193  }
194  else if(predicate_type()(args[sample], this->samples[this->indices[0]]))
195  {
196  std::pop_heap(this->indices.begin(), this->indices.end(), indirect_cmp(this->samples));
197  this->assign(args, this->indices.back());
198  }
199  }
200 
201  result_type result(dont_care) const
202  {
203  if(!this->is_sorted)
204  {
205  // Must use the same predicate here as in push_heap/pop_heap above.
206  std::sort_heap(this->indices.begin(), this->indices.end(), indirect_cmp(this->samples));
207  // sort_heap puts elements in reverse order. Calling std::reverse
208  // turns the sorted sequence back into a valid heap.
209  std::reverse(this->indices.begin(), this->indices.end());
210  this->is_sorted = true;
211  }
212 
213  return detail::make_tail_range(
214  this->samples.begin()
215  , this->indices.begin()
216  , this->indices.end()
217  );
218  }

234  void assign(Args const &args, std::size_t index)
235  {
236  BOOST_ASSERT(index < this->samples.size());
237  this->samples[index] = args[sample];
238  std::push_heap(this->indices.begin(), this->indices.end(), indirect_cmp(this->samples));
239  this->is_sorted = false;
240  // Tell the tail variates to store their values also
241  args[accumulator].template visit_if<is_tail_variate>(detail::stat_assign(args, index));
242  }
