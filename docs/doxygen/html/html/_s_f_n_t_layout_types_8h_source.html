534 /*
535  A BinarySearchHeader defines the five standard fields needed to perform quick
536  lookups in a lookup table (note that using UInt16s, and not ItemCounts or
537  similar types, is important here, since these tables are in fonts, and the
538  documented font formats specify 16-bit quantities).
539 */
540 struct SFNTLookupBinarySearchHeader {
541  UInt16 unitSize; /* size of a unit in bytes */
542  UInt16 nUnits; /* number of units in table */
543  UInt16 searchRange; /* (largest power of two <= nUnits) * unitSize */
544  UInt16 entrySelector; /* log2 (largest power of two <= nUnits) */
545  UInt16 rangeShift; /* (nUnits - largest power of two <= nUnits) * unitSize */
546 };
547 typedef struct SFNTLookupBinarySearchHeader SFNTLookupBinarySearchHeader;
