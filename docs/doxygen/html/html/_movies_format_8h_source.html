191 the incorrect flags is reviewed by developers.
192 enum {
193  kQTSampleDependency_DependsOnOthers = 1<<5, // INCORRECT VALUE
194  kQTSampleDependency_DoesNotDependOnOthers = 1<<4, // INCORRECT VALUE
195  kQTSampleDependency_IsDependedOnByOthers = 1<<3, // INCORRECT VALUE
196  kQTSampleDependency_IsNotDependedOnByOthers = 1<<2, // INCORRECT VALUE
197  kQTSampleDependency_HasRedundantCoding = 1<<1, // INCORRECT VALUE
198  kQTSampleDependency_HasNoRedundantCoding = 1<<0 // INCORRECT VALUE
199 };
200 */

203  /* bit 0x80 is reserved; bit combinations 0x30, 0xC0 and 0x03 are reserved*/

206  kQTSampleDependency_DependsOnOthers_Corrected = 1 << 4, /* ie: not an I picture*/
