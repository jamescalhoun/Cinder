126  atomic_generic_cas32() {}
127  T load(memory_order order=memory_order_seq_cst) const volatile
128  {
129  T expected=(T)i;
130  do { } while(!const_cast<this_type *>(this)->compare_exchange_weak(expected, expected, order, memory_order_relaxed));
131  return expected;
132  }
133  void store(T v, memory_order order=memory_order_seq_cst) volatile
134  {
135  exchange(v);
136  }
137  bool compare_exchange_strong(
138  T &expected,
139  T desired,
140  memory_order success_order,
141  memory_order failure_order) volatile
142  {
143  T found;
144  found=(T)fenced_compare_exchange_strong_32(&i, (int32_t)expected, (int32_t)desired);
145  bool success=(found==expected);
146  expected=found;
147  return success;
148  }
149  bool compare_exchange_weak(
150  T &expected,
151  T desired,
152  memory_order success_order,
153  memory_order failure_order) volatile
154  {
155  return compare_exchange_strong(expected, desired, success_order, failure_order);
156  }
157  T exchange(T r, memory_order order=memory_order_seq_cst) volatile
158  {
159  T expected=(T)i;
160  do { } while(!compare_exchange_weak(expected, r, order, memory_order_relaxed));
161  return expected;
162  }
