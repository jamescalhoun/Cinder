631  pointer linearize() {
632  if (empty())
633  return 0;
634  if (m_first < m_last || m_last == m_buff)
635  return m_first;
636  pointer src = m_first;
637  pointer dest = m_buff;
638  size_type moved = 0;
639  size_type constructed = 0;
640  BOOST_TRY {
641  for (pointer first = m_first; dest < src; src = first) {
642  for (size_type ii = 0; src < m_end; ++src, ++dest, ++moved, ++ii) {
643  if (moved == size()) {
644  first = dest;
645  break;
646  }
647  if (dest == first) {
648  first += ii;
649  break;
650  }
651  if (is_uninitialized(dest)) {
652  m_alloc.construct(dest, *src);
653  ++constructed;
654  } else {
655  value_type tmp = *src;
656  replace(src, *dest);
657  replace(dest, tmp);
658  }
659  }
660  }

662  m_last += constructed;
663  m_size += constructed;
664  BOOST_RETHROW
665  }
