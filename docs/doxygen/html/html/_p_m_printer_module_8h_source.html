75 /*
76 Keys for the printer browser specification dictionary. This CFDictionary,
77 created by the printer module, provides a printer browser module with the
78 information it needs to browse for a given type of printer over its
79 connection type, and to display it in PrintCenter's Browser Window.
80 
81 kPMPrinterBrowserKindKey corresponds to a CFStringRef to a human-readable
82 printer type string that can be displayed in the browser view. For a printer
83 driven by Apple's PostScript printer module over an AppleTalk connection,
84 this would be the string "PostScript printer".
85 
86 kPMPrinterBrowserInfoKey corresponds to a CFTypeRef to connection-specific
87 information used to look up a particular kind of printer over a given IO
88 connection. A PostScript printer module supporting AppleTalk might specify
89 the NBP lookup string "LaserWriter".
90 
91 kPMPrinterBrowserIconKey corresponds to a CFDataRef to the icon family data
92 for the printer icon to be displayed in the browser view. The family data
93 are entirely contained within the CFDataRef; i.e., it is *not* a handle.
94 
95 kPMPrinterBrowserDeviceIDKey corresponds to a CFStringRef to a USB/Firewire
96 IEEE-1284 DeviceID string. e.g., "COMMAND SET: PostScript;" or
97 "MFG: Hewlett-Packard; MDL: DeskJet 935C;"
98 */

123  * Discussion:
124  * Status and error event codes and keys reported by the printer
125  * module (used in PMNotificationProc)
126  */
127 enum {
128 
129  /*
130  * non-error printer status
131  */
132  kPMEventPrinterStatus = 4000,
133 
134  /*
135  * a fatal printing error has occurred. The Printer Module MUST stop
136  * printing and return immediately after sending this event.
137  */
138  kPMEventErrorOccurred = 4001,
139 
140  /*
141  * an auto-recoverable error occurred. The PM SHOULD keep trying to
142  * print the job without waiting for any user action. If the PM
143  * detects that the condition is cleared, it MUST send a
144  * kPMEventRecoverableErrorCleared event, otherwise the condition
145  * alert will not get cleared
146  */
147  kPMEventRecoverableErrorOccurred = 4002,
148 
149  /*
150  * auto-recoverable error has been cleared, the job is continued
151  * automatically and without user intervention. This event causes
152  * the condition alert to close and the job state to change to
153  * "printing".
154  */
155  kPMEventRecoverableErrorCleared = 4003
156 };
