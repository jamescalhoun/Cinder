118 /* A SmallFract value is just the fractional part of a Fixed number,
119 which is the low order word. SmallFracts are used to save room,
120 and to be compatible with Quickdraw's RGBColor. They can be
121 assigned directly to and from INTEGERs. */
122 /* Unsigned fraction between 0 and 1 */
123 typedef unsigned short SmallFract;
124 /* For developmental simplicity in switching between the HLS and HSV
125 models, HLS is reordered into HSL. Thus both models start with
126 hue and saturation values; value/lightness/brightness is last. */
127 
128 struct HSVColor {
129  SmallFract hue; /*Fraction of circle, red at 0*/
130  SmallFract saturation; /*0-1, 0 for gray, 1 for pure color*/
131  SmallFract value; /*0-1, 0 for black, 1 for max intensity*/
132 };
133 typedef struct HSVColor HSVColor;
