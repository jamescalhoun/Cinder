51  };
52  // metrowerks trips this assertion - how come?
53  #if ! defined(__MWERKS__)
54  BOOST_STATIC_ASSERT(128 == sizeof(lookup_table));
55  #endif
56  signed char value = -1;
57  if((unsigned)t <= 127)
58  value = lookup_table[(unsigned)t];
59  if(-1 == value)
60  boost::serialization::throw_exception(
61  dataflow_exception(dataflow_exception::invalid_base64_character)
62  );
63  return value;
64  }
