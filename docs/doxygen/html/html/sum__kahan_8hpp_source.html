55 #endif
56  operator ()(Args const & args)
57  {
58  const Sample myTmp1 = args[parameter::keyword<Tag>::get()] - this->compensation;
59  const Sample myTmp2 = this->sum + myTmp1;
60  this->compensation = (myTmp2 - this->sum) - myTmp1;
61  this->sum = myTmp2;
62  }
