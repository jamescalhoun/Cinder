338  template <typename Reference, typename IndexList, typename TPtr>
339  Reference access_element(boost::type<Reference>,
340  const IndexList& indices,
341  TPtr base,
342  const size_type* extents,
343  const index* strides,
344  const index* index_bases) const {
345  boost::function_requires<
346  CollectionConcept<IndexList> >();
347  ignore_unused_variable_warning(index_bases);
348  ignore_unused_variable_warning(extents);
349 #if !defined(NDEBUG) && !defined(BOOST_DISABLE_ASSERTS)
350  for (size_type i = 0; i != NumDims; ++i) {
351  BOOST_ASSERT(indices[i] - index_bases[i] >= 0);
352  BOOST_ASSERT(size_type(indices[i] - index_bases[i]) < extents[i]);
353  }
354 #endif
355 
356  index offset = 0;
357  {
358  typename IndexList::const_iterator i = indices.begin();
359  size_type n = 0;
360  while (n != NumDims) {
361  offset += (*i) * strides[n];
362  ++n;
363  ++i;
364  }
365  }
366  return base[offset];
367  }

369  template <typename StrideList, typename ExtentList>
370  void compute_strides(StrideList& stride_list, ExtentList& extent_list,
371  const general_storage_order<NumDims>& storage)
372  {
373  // invariant: stride = the stride for dimension n
374  index stride = 1;
375  for (size_type n = 0; n != NumDims; ++n) {
376  index stride_sign = +1;
377 
378  if (!storage.ascending(storage.ordering(n)))
379  stride_sign = -1;
380 
381  // The stride for this dimension is the product of the
382  // lengths of the ranks minor to it.
383  stride_list[storage.ordering(n)] = stride * stride_sign;
384 
385  stride *= extent_list[storage.ordering(n)];
386  }
387  }

389  // This calculates the offset to the array base pointer due to:
390  // 1. dimensions stored in descending order
391  // 2. non-zero dimension index bases
392  template <typename StrideList, typename ExtentList, typename BaseList>
393  index
394  calculate_origin_offset(const StrideList& stride_list,
395  const ExtentList& extent_list,
396  const general_storage_order<NumDims>& storage,
397  const BaseList& index_base_list)

405  // This calculates the offset added to the base pointer that are
406  // caused by descending dimensions
407  template <typename StrideList, typename ExtentList>
408  index
409  calculate_descending_dimension_offset(const StrideList& stride_list,
410  const ExtentList& extent_list,
411  const general_storage_order<NumDims>& storage)
412  {
413  index offset = 0;
414  if (!storage.all_dims_ascending())
415  for (size_type n = 0; n != NumDims; ++n)
416  if (!storage.ascending(n))
417  offset -= (extent_list[n] - 1) * stride_list[n];
418 
419  return offset;
420  }

437  // Slicing using an index_gen.
438  // Note that populating an index_gen creates a type that encodes
439  // both the number of dimensions in the current Array (NumDims), and
440  // the Number of dimensions for the resulting view. This allows the
441  // compiler to fail if the dimensions aren't completely accounted
442  // for. For reasons unbeknownst to me, a BOOST_STATIC_ASSERT
443  // within the member function template does not work. I should add a
444  // note to the documentation specifying that you get a damn ugly
445  // error message if you screw up in your slicing code.
446  template <typename ArrayRef, int NDims, typename TPtr>
447  ArrayRef
448  generate_array_view(boost::type<ArrayRef>,
449  const boost::detail::multi_array::
450  index_gen<NumDims,NDims>& indices,
451  const size_type* extents,
452  const index* strides,
453  const index* index_bases,
454  TPtr base) const {
455 
456  boost::array<index,NDims> new_strides;
457  boost::array<index,NDims> new_extents;
458 
459  index offset = 0;
460  size_type dim = 0;
461  for (size_type n = 0; n != NumDims; ++n) {
462 
463  // Use array specs and input specs to produce real specs.
464  const index default_start = index_bases[n];
465  const index default_finish = default_start+extents[n];
466  const index_range& current_range = indices.ranges_[n];
467  index start = current_range.get_start(default_start);
468  index finish = current_range.get_finish(default_finish);
469  index stride = current_range.stride();
470  BOOST_ASSERT(stride != 0);
471 
472  // An index range indicates a half-open strided interval
473  // [start,finish) (with stride) which faces upward when stride
474  // is positive and downward when stride is negative,
475 
476  // RG: The following code for calculating length suffers from
477  // some representation issues: if finish-start cannot be represented as
478  // by type index, then overflow may result.
479 
480  index len;
481  if ((finish - start) / stride < 0) {
482  // [start,finish) is empty according to the direction imposed by
483  // the stride.
484  len = 0;
485  } else {
486  // integral trick for ceiling((finish-start) / stride)
487  // taking into account signs.
488  index shrinkage = stride > 0 ? 1 : -1;
489  len = (finish - start + (stride - shrinkage)) / stride;
490  }
491 
492  // start marks the closed side of the range, so it must lie
493  // exactly in the set of legal indices
494  // with a special case for empty arrays
495  BOOST_ASSERT(index_bases[n] <= start &&
496  ((start <= index_bases[n]+index(extents[n])) ||
497  (start == index_bases[n] && extents[n] == 0)));
498 
499 #ifndef BOOST_DISABLE_ASSERTS
500  // finish marks the open side of the range, so it can go one past
501  // the "far side" of the range (the top if stride is positive, the bottom
502  // if stride is negative).
503  index bound_adjustment = stride < 0 ? 1 : 0;
504  BOOST_ASSERT(((index_bases[n] - bound_adjustment) <= finish) &&
505  (finish <= (index_bases[n] + index(extents[n]) - bound_adjustment)));
506 #endif // BOOST_DISABLE_ASSERTS
507 
508 
509  // the array data pointer is modified to account for non-zero
510  // bases during slicing (see [Garcia] for the math involved)
511  offset += start * strides[n];
512 
513  if (!current_range.is_degenerate()) {
514 
515  // The stride for each dimension is included into the
516  // strides for the array_view (see [Garcia] for the math involved).
517  new_strides[dim] = stride * strides[n];
518 
519  // calculate new extents
520  new_extents[dim] = len;
521  ++dim;
522  }
523  }
524  BOOST_ASSERT(dim == NDims);
525 
526  return
527  ArrayRef(base+offset,
528  new_extents,
529  new_strides);
530  }
