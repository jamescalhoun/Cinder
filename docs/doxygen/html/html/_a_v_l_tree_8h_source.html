52 /*
53  * AVLTree
54  *
55  * Discussion:
56  * Prototypes for routines which create, destroy, allow for
57  * insertion, deleting, and iteration of routines in an AVL balanced
58  * binary tree.
59  *
60  * An AVL tree is a balanced, binary tree which is fairly fast for
61  * finds and acceptably fast for insertion and deletion. The tree
62  * is kept balanced, so that the heights of any given node's left
63  * and right branches never differ by more than 1, which keeps
64  * performance from being too horribe in the degenerate case.
65  *
66  *
67  * Very loosely based on some public domain source code for doing
68  * avl trees and on the discussion in Sedgewick "Algorithms" book.
69  */
70 typedef UInt16 AVLVisitStage;
71 enum {
72  kAVLPreOrder = 0,
73  kAVLInOrder = 1,
74  kAVLPostOrder = 2
75 };

109 /*
110  Every tree must have a function which compares the data for two items and returns < 0, 0, or >0
111  for the items - < 0 if the first item is 'before' the second item according to some criteria,
112  == 0 if the two items are identical according to the criteria, or > 0 if the first item is
113  'after' the second item according to the criteria. The comparison function is also passed the
114  node type, but most of the time this can be ignored.
115 */
116 typedef CALLBACK_API( SInt32 , AVLCompareItemsProcPtr )(AVLTreePtr tree, const void *i1, const void *i2, AVLNodeType nd_typ);
117 /*
118  Every tree must have a itemSizeProc; this routine gets passed a pointer to the item's data and
119  returns the size of the data. If a tree contains records of a fixed size, this function can
120  just return sizeof( that-struct ); otherwise it should calculate the size of the item based on
121  the data for the item.
122 */
123 typedef CALLBACK_API( UInt32 , AVLItemSizeProcPtr )(AVLTreePtr tree, const void *itemPtr);
124 /*
125  A tree may have an optional disposeItemProc, which gets called whenever an item is removed
126  from the tree ( via AVLRemove() or when AVLDispose() deletes all of the items in the tree ).
127  This might be useful if the nodes in the tree own 'resources' ( like, open files ) which
128  should be released before the item is removed.
129 */
130 typedef CALLBACK_API( void , AVLDisposeItemProcPtr )(AVLTreePtr tree, const void *dataP);
131 /*
132  The common way to iterate across all of the items in a tree is via AVLWalk(), which takes
133  a walkProcPtr. This function will get called for every item in the tree three times, as
134  the tree is being walked across. First, the walkProc will get called with visitStage ==
135  kAVLPreOrder, at which point internally the node of the tree for the given data has just
136  been reached. Later, this function will get called with visitStage == kAVLInOrder, and
137  lastly this function will get called with visitStage == kAVLPostOrder.
138  The 'minimum' item in the tree will get called with visitStage == kInOrder first, followed
139  by the 'next' item in the tree, up until the last item in the tree structure is called.
140  In general, you'll only care about calls to this function when visitStage == kAVLInOrder.
141 */
142 typedef CALLBACK_API( OSErr , AVLWalkProcPtr )(AVLTreePtr tree, const void *dataP, AVLVisitStage visitStage, AVLNodeType node, UInt32 level, SInt32 balance, void *refCon);

442 /*
443  Iterate across all of the items in the tree, in the order specified. kLeftToRight is
444  basically lowest-to-highest order, kRightToLeft is highest-to-lowest order. For each
445  node in the tree, it will call the walkProc with three messages ( at the appropriate
446  time ). First, with kAVLPreOrder when the walking gets to this node in the tree,
447  before handling either the left or right subtree, secondly, with kAVLInOrder after
448  handling one subtree but before handling the other, and lastly with kAVLPostOrder after
449  handling both subtrees. If you want to handle items in order, then only do something
450  if the visit stage is kAVLInOrder. You can only call AVLRemove() from inside a walkProc
451  if visit stage is kAVLPostOrder ( because if you remove a node during the pre or in order
452  stages you will corrupt the list ) OR if you return a non-zero result from the walkProc
453  call which called AVLRemove() to immediately terminate the walkProc. Do not call AVLInsert()
454  to insert a node into the tree from inside a walkProc.
455  The walkProc function gets called with the AVLTreePtr, a pointer to the data for the
456  current node ( which you can change in place as long as you do not affect the order within
457  the tree ), the visit stage, the type of the current node ( leaf node, right or left branch,
458  or full tree ), the level within the tree ( the root is level 1 ), the balance for the
459  current node, and the refCon passed to AVLWalk(). This refCon is different from the one passed
460  into AVLInit(); use AVLGetRefCon() to get that refCon if you want it inside a walkProc.
461  ( Most walkProcs will not care about the values for node type, level, or balance. )
462 */
463 /*
464  * AVLWalk()
465  *
466  * Availability:

468  * CarbonLib: in CarbonLib 1.0 and later
469  * Mac OS X: in version 10.0 and later
470  */
