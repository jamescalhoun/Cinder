118 template
119 <
120  typename Range1,
121  typename Range2,
122  bool Reverse = false
123 >
124 struct range_to_range
125 {
126  typedef typename reversible_view
127  <
128  Range1 const,
129  Reverse ? iterate_reverse : iterate_forward
130  >::type rview_type;
131  typedef typename closeable_view
132  <
133  rview_type const,
134  geometry::closure<Range1>::value
135  >::type view_type;
136 
137  static inline void apply(Range1 const& source, Range2& destination)
138  {
139  geometry::clear(destination);
140 
141  rview_type rview(source);
142 
143  // We consider input always as closed, and skip last
144  // point for open output.
145  view_type view(rview);
146 
147  int n = boost::size(view);
148  if (geometry::closure<Range2>::value == geometry::open)
149  {
150  n--;
151  }
152 
153  int i = 0;
154  for (typename boost::range_iterator<view_type const>::type it
155  = boost::begin(view);
156  it != boost::end(view) && i < n;
157  ++it, ++i)
158  {
159  geometry::append(destination, *it);
160  }
161  }
162 };

164 template <typename Polygon1, typename Polygon2>
165 struct polygon_to_polygon
166 {
167  typedef range_to_range
168  <
169  typename geometry::ring_type<Polygon1>::type,
170  typename geometry::ring_type<Polygon2>::type,
171  geometry::point_order<Polygon1>::value
172  != geometry::point_order<Polygon2>::value
173  > per_ring;
174 
175  static inline void apply(Polygon1 const& source, Polygon2& destination)
176  {
177  // Clearing managed per ring, and in the resizing of interior rings
178 
179  per_ring::apply(geometry::exterior_ring(source),
180  geometry::exterior_ring(destination));
181 
182  // Container should be resizeable
183  traits::resize
184  <
185  typename boost::remove_reference
186  <
187  typename traits::interior_mutable_type<Polygon2>::type
188  >::type
189  >::apply(interior_rings(destination), num_interior_rings(source));
190 
191  typename interior_return_type<Polygon1 const>::type rings_source
192  = interior_rings(source);
193  typename interior_return_type<Polygon2>::type rings_dest
194  = interior_rings(destination);
195 
196  BOOST_AUTO_TPL(it_source, boost::begin(rings_source));
197  BOOST_AUTO_TPL(it_dest, boost::begin(rings_dest));
198 
199  for ( ; it_source != boost::end(rings_source); ++it_source, ++it_dest)
200  {
201  per_ring::apply(*it_source, *it_dest);
202  }
203  }
204 };
