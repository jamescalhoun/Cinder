19 // iterator which takes elements of x bits and returns elements of y bits.
20 // used to change streams of 8 bit characters into streams of 6 bit characters.
21 // and vice-versa for implementing base64 encodeing/decoding. Be very careful
22 // when using and end iterator. end is only reliable detected when the input
23 // stream length is some common multiple of x and y. E.G. Base64 6 bit
24 // character and 8 bit bytes. Lowest common multiple is 24 => 4 6 bit characters

134 void transform_width<Base, BitsOut, BitsIn, CharType>::fill() {
135  unsigned int missing_bits = BitsOut;
136  m_buffer_out = 0;
137  do{
138  if(0 == m_remaining_bits){
139  if(m_end_of_sequence){
140  m_buffer_in = 0;
141  m_remaining_bits = missing_bits;
142  }
143  else{
144  m_buffer_in = * this->base_reference()++;
145  m_remaining_bits = BitsIn;
146  }
147  }
148 
149  // append these bits to the next output
150  // up to the size of the output
151  unsigned int i = std::min(missing_bits, m_remaining_bits);
152  // shift interesting bits to least significant position
153  base_value_type j = m_buffer_in >> (m_remaining_bits - i);
154  // and mask off the un interesting higher bits
155  // note presumption of twos complement notation
156  j &= (1 << i) - 1;
157  // append then interesting bits to the output value
158  m_buffer_out <<= i;
159  m_buffer_out |= j;
160 
161  // and update counters
162  missing_bits -= i;
163  m_remaining_bits -= i;
164  }while(0 < missing_bits);
165  m_buffer_out_full = true;
166 }
