462 //This function uses binary search to discover the
463 //highest set bit of the integer
464 inline std::size_t floor_log2 (std::size_t x)
465 {
466  const std::size_t Bits = sizeof(std::size_t)*CHAR_BIT;
467  const bool Size_t_Bits_Power_2= !(Bits & (Bits-1));
468  BOOST_STATIC_ASSERT(Size_t_Bits_Power_2);
469 
470  std::size_t n = x;
471  std::size_t log2 = 0;
472 
473  for(std::size_t shift = Bits >> 1; shift; shift >>= 1){
474  std::size_t tmp = n >> shift;
475  if (tmp)
476  log2 += shift, n = tmp;
477  }
478 
479  return log2;
480 }

482 inline float fast_log2 (float val)
483 {
484  union caster_t
485  {
486  boost::uint32_t x;
487  float val;
488  } caster;
489 
490  caster.val = val;
491  boost::uint32_t x = caster.x;
492  const int log_2 = (int)(((x >> 23) & 255) - 128);
493  x &= ~(255 << 23);
494  x += 127 << 23;
495  caster.x = x;
496  val = caster.val;
497  val = ((-1.0f/3.f) * val + 2.f) * val - (2.0f/3.f);
498 
499  return (val + log_2);
500 }
