135  void_pointer incorporate_after(iterator after_this, const void_pointer &b, size_type unit_bytes, size_type num_units)
136  {
137  typedef typename boost::intrusive::pointer_traits<char_ptr> char_pointer_traits;
138  char_ptr elem = char_pointer_traits::static_cast_from(b);
139  if(num_units){
140  char_ptr prev_elem = elem;
141  elem += unit_bytes;
142  for(size_type i = 0; i != num_units-1; ++i, elem += unit_bytes){
143  ::new (container_detail::to_raw_pointer(prev_elem)) void_pointer(elem);
144  prev_elem = elem;
145  }
146  slist_impl_.incorporate_after(after_this, to_node_ptr(b), to_node_ptr(prev_elem), num_units);
147  }
148  return elem;
149  }
