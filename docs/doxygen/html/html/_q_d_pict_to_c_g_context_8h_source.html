38 typedef struct QDPict* QDPictRef;
39 /*
40  Note: QuickDraw picture data typically comes in two forms: a PICT resource
41  that begins the picture header data at the beginning of the resource and PICT
42  files that begin with 512 bytes of arbitrary data, followed by
43  the picture header data. For this reason, the routines that create a QDPictRef
44  attempt to find the picture header data beginning at either the first byte
45  of the data provided or at byte 513 of the data provided.
46 
47  Additionally the Picture Bounds must not be an empty rect.
48 */
49 /* Create a QDPict reference, using `provider' to obtain the QDPict's data.
50  * It is assumed that either the first byte or the 513th byte of data
51  * in the file referenced by the URL is the first byte of the
52  * picture header. If the URL does not begin PICT data at one
53  * of these places in the data fork then the QDPictRef returned will be NULL.
54 */
55 /*
56  * QDPictCreateWithProvider()
57  *
58  * Availability:
59  * Non-Carbon CFM: not available
60  * CarbonLib: not available
61  * Mac OS X: in version 10.1 and later
62  */
63 EXTERN_API_C( QDPictRef )
64 QDPictCreateWithProvider(CGDataProviderRef provider);
