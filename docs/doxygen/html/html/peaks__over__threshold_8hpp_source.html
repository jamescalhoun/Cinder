52  // peaks_over_threshold_impl
53  // works with an explicit threshold value and does not depend on order statistics
120  template<typename Sample, typename LeftRight>
121  struct peaks_over_threshold_impl
122  : accumulator_base
123  {
124  typedef typename numeric::functional::average<Sample, std::size_t>::result_type float_type;
125  // for boost::result_of
126  typedef boost::tuple<float_type, float_type, float_type> result_type;
127  // for left tail fitting, mirror the extreme values
128  typedef mpl::int_<is_same<LeftRight, left>::value ? -1 : 1> sign;
129 
130  template<typename Args>
131  peaks_over_threshold_impl(Args const &args)
132  : Nu_(0)
133  , mu_(sign::value * numeric::average(args[sample | Sample()], (std::size_t)1))
134  , sigma2_(numeric::average(args[sample | Sample()], (std::size_t)1))
135  , threshold_(sign::value * args[pot_threshold_value])
136  , fit_parameters_(boost::make_tuple(0., 0., 0.))
137  , is_dirty_(true)
138  {
139  }
140 
141  template<typename Args>
142  void operator ()(Args const &args)
143  {
144  this->is_dirty_ = true;
145 
146  if (sign::value * args[sample] > this->threshold_)
147  {
148  this->mu_ += args[sample];
149  this->sigma2_ += args[sample] * args[sample];
150  ++this->Nu_;
151  }
152  }
153 
154  template<typename Args>
155  result_type result(Args const &args) const
156  {
157  if (this->is_dirty_)
158  {
159  this->is_dirty_ = false;
160 
161  std::size_t cnt = count(args);
162 
163  this->mu_ = sign::value * numeric::average(this->mu_, this->Nu_);
164  this->sigma2_ = numeric::average(this->sigma2_, this->Nu_);
165  this->sigma2_ -= this->mu_ * this->mu_;
166 
167  float_type threshold_probability = numeric::average(cnt - this->Nu_, cnt);
168 
169  float_type tmp = numeric::average(( this->mu_ - this->threshold_ )*( this->mu_ - this->threshold_ ), this->sigma2_);
170  float_type xi_hat = 0.5 * ( 1. - tmp );
171  float_type beta_hat = 0.5 * ( this->mu_ - this->threshold_ ) * ( 1. + tmp );
172  float_type beta_bar = beta_hat * std::pow(1. - threshold_probability, xi_hat);
173  float_type u_bar = this->threshold_ - beta_bar * ( std::pow(1. - threshold_probability, -xi_hat) - 1.)/xi_hat;
174  this->fit_parameters_ = boost::make_tuple(u_bar, beta_bar, xi_hat);
175  }
176 
177  return this->fit_parameters_;
178  }
179 
180  private:
181  std::size_t Nu_; // number of samples larger than threshold
182  mutable float_type mu_; // mean of Nu_ largest samples
183  mutable float_type sigma2_; // variance of Nu_ largest samples
184  float_type threshold_;
185  mutable result_type fit_parameters_; // boost::tuple that stores fit parameters
186  mutable bool is_dirty_;
187  };

190  // peaks_over_threshold_prob_impl
191  // determines threshold from a given threshold probability using order statistics
200  template<typename Sample, typename LeftRight>
201  struct peaks_over_threshold_prob_impl
202  : accumulator_base
203  {
204  typedef typename numeric::functional::average<Sample, std::size_t>::result_type float_type;
205  // for boost::result_of
206  typedef boost::tuple<float_type, float_type, float_type> result_type;
207  // for left tail fitting, mirror the extreme values
208  typedef mpl::int_<is_same<LeftRight, left>::value ? -1 : 1> sign;
209 
210  template<typename Args>
211  peaks_over_threshold_prob_impl(Args const &args)
212  : mu_(sign::value * numeric::average(args[sample | Sample()], (std::size_t)1))
213  , sigma2_(numeric::average(args[sample | Sample()], (std::size_t)1))
214  , threshold_probability_(args[pot_threshold_probability])
215  , fit_parameters_(boost::make_tuple(0., 0., 0.))
216  , is_dirty_(true)
217  {
218  }
219 
220  void operator ()(dont_care)
221  {
222  this->is_dirty_ = true;
223  }
224 
225  template<typename Args>
226  result_type result(Args const &args) const
227  {
228  if (this->is_dirty_)
229  {
230  this->is_dirty_ = false;
231 
232  std::size_t cnt = count(args);
233 
234  // the n'th cached sample provides an approximate threshold value u
235  std::size_t n = static_cast<std::size_t>(
236  std::ceil(
237  cnt * ( ( is_same<LeftRight, left>::value ) ? this->threshold_probability_ : 1. - this->threshold_probability_ )
238  )
239  );
240 
241  // If n is in a valid range, return result, otherwise return NaN or throw exception
242  if ( n >= static_cast<std::size_t>(tail(args).size()))
243  {
244  if (std::numeric_limits<float_type>::has_quiet_NaN)
245  {
246  return boost::make_tuple(
247  std::numeric_limits<float_type>::quiet_NaN()
248  , std::numeric_limits<float_type>::quiet_NaN()
249  , std::numeric_limits<float_type>::quiet_NaN()
250  );
251  }
252  else
253  {
254  std::ostringstream msg;
255  msg << "index n = " << n << " is not in valid range [0, " << tail(args).size() << ")";
256  boost::throw_exception(std::runtime_error(msg.str()));
257  return boost::make_tuple(Sample(0), Sample(0), Sample(0));
258  }
259  }
260  else
261  {
262  float_type u = *(tail(args).begin() + n - 1) * sign::value;
263 
264  // compute mean and variance of samples above/under threshold value u
265  for (std::size_t i = 0; i < n; ++i)
266  {
267  mu_ += *(tail(args).begin() + i);
268  sigma2_ += *(tail(args).begin() + i) * (*(tail(args).begin() + i));
269  }
270 
271  this->mu_ = sign::value * numeric::average(this->mu_, n);
272  this->sigma2_ = numeric::average(this->sigma2_, n);
273  this->sigma2_ -= this->mu_ * this->mu_;
274 
275  if (is_same<LeftRight, left>::value)
276  this->threshold_probability_ = 1. - this->threshold_probability_;
277 
278  float_type tmp = numeric::average(( this->mu_ - u )*( this->mu_ - u ), this->sigma2_);
279  float_type xi_hat = 0.5 * ( 1. - tmp );
280  float_type beta_hat = 0.5 * ( this->mu_ - u ) * ( 1. + tmp );
281  float_type beta_bar = beta_hat * std::pow(1. - threshold_probability_, xi_hat);
282  float_type u_bar = u - beta_bar * ( std::pow(1. - threshold_probability_, -xi_hat) - 1.)/xi_hat;
283  this->fit_parameters_ = boost::make_tuple(u_bar, beta_bar, xi_hat);
284  }
285  }
286 
287  return this->fit_parameters_;
288  }
289 
290  private:
291  mutable float_type mu_; // mean of samples above threshold u
292  mutable float_type sigma2_; // variance of samples above threshold u
293  mutable float_type threshold_probability_;
294  mutable result_type fit_parameters_; // boost::tuple that stores fit parameters
295  mutable bool is_dirty_;
296  };
