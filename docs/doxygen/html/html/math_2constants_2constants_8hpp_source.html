30  // To permit other calculations at about 100 decimal digits with NTL::RR type,
31  // it is obviously necessary to define constants to this accuracy.
32 
33  // However, some compilers do not accept decimal digits strings as long as this.
34  // So the constant is split into two parts, with the 1st containing at least
35  // long double precision, and the 2nd zero if not needed or known.
36  // The 3rd part permits an exponent to be provided if necessary (use zero if none) -
37  // the other two parameters may only contain decimal digits (and sign and decimal point),
38  // and may NOT include an exponent like 1.234E99.
39  // The second digit string is only used if T is a User-Defined Type,
40  // when the constant is converted to a long string literal and lexical_casted to type T.
41  // (This is necessary because you can't use a numeric constant
42  // since even a long double might not have enough digits).
43 
44  enum construction_method
45  {
46  construct_from_float = 1,
47  construct_from_double = 2,
48  construct_from_long_double = 3,
49  construct_from_string = 4
50  };

52  //
53  // Max number of binary digits in the string representations
54  // of our constants:
55  //
56  BOOST_STATIC_CONSTANT(int, max_string_digits = (101 * 1000L) / 301L);

58  template <class Real, class Policy>
59  struct construction_traits
60  {
61  private:
62  typedef typename policies::precision<Real, Policy>::type t1;
63  typedef typename policies::precision<float, Policy>::type t2;
64  typedef typename policies::precision<double, Policy>::type t3;
65  typedef typename policies::precision<long double, Policy>::type t4;
66  public:
67  typedef typename mpl::if_<
68  mpl::and_<boost::is_convertible<float, Real>, mpl::bool_< t1::value <= t2::value>, mpl::bool_<0 != t1::value> >,
69  mpl::int_<construct_from_float>,
70  typename mpl::if_<
71  mpl::and_<boost::is_convertible<double, Real>, mpl::bool_< t1::value <= t3::value>, mpl::bool_<0 != t1::value> >,
72  mpl::int_<construct_from_double>,
73  typename mpl::if_<
74  mpl::and_<boost::is_convertible<long double, Real>, mpl::bool_< t1::value <= t4::value>, mpl::bool_<0 != t1::value> >,
75  mpl::int_<construct_from_long_double>,
76  typename mpl::if_<
77  mpl::and_<mpl::bool_< t1::value <= max_string_digits>, mpl::bool_<0 != t1::value> >,
78  mpl::int_<construct_from_string>,
79  mpl::int_<t1::value>
80  >::type
81  >::type
82  >::type
83  >::type type;
84  };

161  #define BOOST_DEFINE_MATH_CONSTANT(name, x, y)\
162  namespace detail{\
163  template <class T> struct BOOST_JOIN(constant_, name){\
164  private:\
165  /* The default implementations come next: */ \
166  static inline T get_from_string()\
167  {\
168  static const T result = convert_from_string<T>(y, boost::is_convertible<const char*, T>());\
169  return result;\
170  }\
171  /* This one is for very high precision that is none the less known at compile time: */ \
172  template <int N> static T compute(BOOST_MATH_EXPLICIT_TEMPLATE_TYPE_SPEC(mpl::int_<N>));\
173  /* public getters come next */\
174  public:\
175  static inline T get(const mpl::int_<construct_from_string>&)\
176  {\
177  constant_initializer<T, & BOOST_JOIN(constant_, name)<T>::get_from_string >::force_instantiate();\
178  return get_from_string();\
179  }\
180  static inline BOOST_CONSTEXPR T get(const mpl::int_<construct_from_float>)\
181  { return BOOST_JOIN(x, F); }\
182  static inline BOOST_CONSTEXPR T get(const mpl::int_<construct_from_double>&)\
183  { return x; }\
184  static inline BOOST_CONSTEXPR T get(const mpl::int_<construct_from_long_double>&)\
185  { return BOOST_JOIN(x, L); }\
186  template <int N> static inline T get(const mpl::int_<N>&)\
187  {\
188  constant_initializer2<T, N, & BOOST_JOIN(constant_, name)<T>::template compute<N> >::force_instantiate();\
189  return compute<N>(); \
190  }\
191  /* This one is for true arbitary precision, which may well vary at runtime: */ \
192  static inline T get(const mpl::int_<0>&)\
193  { return tools::digits<T>() > max_string_digits ? compute<0>() : get(mpl::int_<construct_from_string>()); }\
194  }; /* end of struct */\
195  } /* namespace detail */ \
196  \
197  \
