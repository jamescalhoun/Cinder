98 template <typename Point, std::size_t Dimension, std::size_t DimensionCount>
99 struct parsing_assigner
100 {
101  static inline void apply(tokenizer::iterator& it, tokenizer::iterator end,
102  Point& point, std::string const& wkt)
103  {
104  typedef typename coordinate_type<Point>::type coordinate_type;
105 
106  // Stop at end of tokens, or at "," ot ")"
107  bool finished = (it == end || *it == "," || *it == ")");
108 
109  try
110  {
111  // Initialize missing coordinates to default constructor (zero)
112  // OR
113  // Use lexical_cast for conversion to double/int
114  // Note that it is much slower than atof. However, it is more standard
115  // and in parsing the change in performance falls probably away against
116  // the tokenizing
117  set<Dimension>(point, finished
118  ? coordinate_type()
119  : coordinate_cast<coordinate_type>::apply(*it));
120  }
121  catch(boost::bad_lexical_cast const& blc)
122  {
123  throw read_wkt_exception(blc.what(), it, end, wkt);
124  }
125  catch(std::exception const& e)
126  {
127  throw read_wkt_exception(e.what(), it, end, wkt);
128  }
129  catch(...)
130  {
131  throw read_wkt_exception("", it, end, wkt);
132  }
133 
134  parsing_assigner<Point, Dimension + 1, DimensionCount>::apply(
135  (finished ? it : ++it), end, point, wkt);
136  }
137 };

151 inline void handle_open_parenthesis(Iterator& it,
152  Iterator const& end, std::string const& wkt)
153 {
154  if (it == end || *it != "(")

194 {
195  // Version with output iterator
196  template <typename OutputIterator>
197  static inline void apply(tokenizer::iterator& it, tokenizer::iterator end,
198  std::string const& wkt, OutputIterator out)
199  {
200  handle_open_parenthesis(it, end, wkt);

202  Point point;

204  // Parse points until closing parenthesis

316 struct polygon_parser

321  static inline void apply(tokenizer::iterator& it, tokenizer::iterator end,
322  std::string const& wkt, Polygon& poly)
323  {
324 
325  handle_open_parenthesis(it, end, wkt);

385 inline void handle_empty_z_m(tokenizer::iterator& it, tokenizer::iterator end,
386  bool& has_empty, bool& has_z, bool& has_m)
387 {
388  has_empty = false;
389  has_z = false;
390  has_m = false;
391 
392  // WKT can optionally have Z and M (measured) values as in
393  // POINT ZM (1 1 5 60), POINT M (1 1 80), POINT Z (1 1 5)
394  // GGL supports any of them as coordinate values, but is not aware
395  // of any Measured value.
396  while (it != end
397  && (one_of(it, "M", has_m)
398  || one_of(it, "Z", has_z)
399  || one_of(it, "EMPTY", has_empty)
400  || one_of(it, "MZ", has_m, has_z)
401  || one_of(it, "ZM", has_z, has_m)
402  )
403  )
404  {
405  ++it;
406  }
407 }

414 template <typename Geometry>
415 inline bool initialize(tokenizer const& tokens,
416  std::string const& geometry_name, std::string const& wkt,
417  tokenizer::iterator& it)
418 {
419  it = tokens.begin();
420  if (it != tokens.end() && boost::iequals(*it++, geometry_name))
421  {
422  bool has_empty, has_z, has_m;
423 
424  handle_empty_z_m(it, tokens.end(), has_empty, has_z, has_m);
425 
426  if (has_z && dimension<Geometry>::type::value < 3)
427  {
428  throw read_wkt_exception("Z only allowed for 3 or more dimensions", wkt);
429  }
430  if (has_empty)
431  {
432  check_end(it, tokens.end(), wkt);
433  return false;
434  }
435  // M is ignored at all.
436 
437  return true;
438  }
439  throw read_wkt_exception(std::string("Should start with '") + geometry_name + "'", wkt);
440 }

443 template <typename Geometry, template<typename> class Parser, typename PrefixPolicy>
444 struct geometry_parser
445 {
446  static inline void apply(std::string const& wkt, Geometry& geometry)
447  {
448  geometry::clear(geometry);
449 
450  tokenizer tokens(wkt, boost::char_separator<char>(" ", ",()"));
451  tokenizer::iterator it;
452  if (initialize<Geometry>(tokens, PrefixPolicy::apply(), wkt, it))

472 template <typename Box>
473 struct box_parser
474 {
475  static inline void apply(std::string const& wkt, Box& box)
476  {
477  bool should_close = false;
478  tokenizer tokens(wkt, boost::char_separator<char>(" ", ",()"));
479  tokenizer::iterator it = tokens.begin();
480  tokenizer::iterator end = tokens.end();
481  if (it != end && boost::iequals(*it, "POLYGON"))
482  {
483  ++it;
484  bool has_empty, has_z, has_m;
485  handle_empty_z_m(it, end, has_empty, has_z, has_m);
486  if (has_empty)
487  {
488  assign_zero(box);
489  return;
490  }
491  handle_open_parenthesis(it, end, wkt);
492  should_close = true;
493  }
494  else if (it != end && boost::iequals(*it, "BOX"))
495  {
496  ++it;
497  }
498  else
499  {
500  throw read_wkt_exception("Should start with 'POLYGON' or 'BOX'", wkt);
501  }
502 
503  typedef typename point_type<Box>::type point_type;
504  std::vector<point_type> points;
505  container_inserter<point_type>::apply(it, end, wkt, std::back_inserter(points));
506 
507  if (should_close)
508  {
509  handle_close_parenthesis(it, end, wkt);
510  }
511  check_end(it, end, wkt);
512 
513  int index = 0;
514  int n = boost::size(points);
515  if (n == 2)
516  {
517  index = 1;
518  }
519  else if (n == 4 || n == 5)
520  {
521  // In case of 4 or 5 points, we do not check the other ones, just
522  // take the opposite corner which is always 2
523  index = 2;
524  }
525  else
526  {
527  throw read_wkt_exception("Box should have 2,4 or 5 points", wkt);
528  }
529 
530  geometry::detail::assign_point_to_index<min_corner>(points.front(), box);
531  geometry::detail::assign_point_to_index<max_corner>(points[index], box);
532  }
533 };
