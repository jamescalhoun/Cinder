1249  /*
1250  \post If the capacity is low it will be increased.
1251  */
1252  void check_low_capacity(size_type n = 1) {
1253  size_type new_size = size() + n;
1254  size_type new_capacity = circular_buffer<T, Alloc>::capacity();
1255  if (new_size > new_capacity) {
1256  if (new_capacity == 0)
1257  new_capacity = 1;
1258  for (; new_size > new_capacity; new_capacity *= 2) {}
1259  circular_buffer<T, Alloc>::set_capacity(
1260  ensure_reserve(new_capacity, new_size));
1261  }
1262 #if BOOST_CB_ENABLE_DEBUG
1263  this->invalidate_iterators_except(end());
1264 #endif
1265  }
1266 
1268  /*
1269  \post If the capacity is high it will be decreased.
1270  */
1271  void check_high_capacity() {
1272  size_type new_capacity = circular_buffer<T, Alloc>::capacity();
1273  while (new_capacity / 3 >= size()) { // (new_capacity / 3) -> avoid oscillations
1274  new_capacity /= 2;
1275  if (new_capacity <= m_capacity_ctrl.min_capacity()) {
1276  new_capacity = m_capacity_ctrl.min_capacity();
1277  break;
1278  }
1279  }
1280  circular_buffer<T, Alloc>::set_capacity(
1281  ensure_reserve(new_capacity, size()));
1282 #if BOOST_CB_ENABLE_DEBUG
1283  this->invalidate_iterators_except(end());
1284 #endif
1285  }
