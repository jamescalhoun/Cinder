97  void allocate_nodes(const size_type n, multiallocation_chain &chain)
98  {
99  //Preallocate all needed blocks to fulfill the request
100  size_type cur_nodes = m_freelist.size();
101  if(cur_nodes < n){
102  this->priv_alloc_block(((n - cur_nodes) - 1)/m_nodes_per_block + 1);
103  }
104 
105  //We just iterate the needed nodes to get the last we'll erase
106  typedef typename free_nodes_t::iterator free_iterator;
107  free_iterator before_last_new_it = m_freelist.before_begin();
108  for(size_type j = 0; j != n; ++j){
109  ++before_last_new_it;
110  }
111 
112  //Cache the first node of the allocated range before erasing
113  free_iterator first_node(m_freelist.begin());
114  free_iterator last_node (before_last_new_it);
115 
116  //Erase the range. Since we already have the distance, this is O(1)
117  m_freelist.erase_after( m_freelist.before_begin()
118  , ++free_iterator(before_last_new_it)
119  , n);
120 
121  //Now take the last erased node and just splice it in the end
122  //of the intrusive list that will be traversed by the multialloc iterator.
123  chain.incorporate_after(chain.before_begin(), &*first_node, &*last_node, n);
124  m_allocated += n;
125  }

139  void deallocate_free_blocks()
140  {
141  typedef typename free_nodes_t::iterator nodelist_iterator;
142  typename blockslist_t::iterator bit(m_blocklist.before_begin()),
143  it(m_blocklist.begin()),
144  itend(m_blocklist.end());
145  free_nodes_t backup_list;
146  nodelist_iterator backup_list_last = backup_list.before_begin();
147 
148  //Execute the algorithm and get an iterator to the last value
149  size_type blocksize = get_rounded_size
150  (m_real_node_size*m_nodes_per_block, (size_type) alignment_of<node_t>::value);
151 
152  while(it != itend){
153  //Collect all the nodes from the block pointed by it
154  //and push them in the list
155  free_nodes_t free_nodes;
156  nodelist_iterator last_it = free_nodes.before_begin();
157  const void *addr = get_block_from_hook(&*it, blocksize);
158 
159  m_freelist.remove_and_dispose_if
160  (is_between(addr, blocksize), push_in_list(free_nodes, last_it));
161 
162  //If the number of nodes is equal to m_nodes_per_block
163  //this means that the block can be deallocated
164  if(free_nodes.size() == m_nodes_per_block){
165  //Unlink the nodes
166  free_nodes.clear();
167  it = m_blocklist.erase_after(bit);
168  mp_segment_mngr_base->deallocate((void*)addr);
169  }
170  //Otherwise, insert them in the backup list, since the
171  //next "remove_if" does not need to check them again.
172  else{
173  //Assign the iterator to the last value if necessary
174  if(backup_list.empty() && !m_freelist.empty()){
175  backup_list_last = last_it;
176  }
177  //Transfer nodes. This is constant time.
178  backup_list.splice_after
179  ( backup_list.before_begin()
180  , free_nodes
181  , free_nodes.before_begin()
182  , last_it
183  , free_nodes.size());
184  bit = it;
185  ++it;
186  }
187  }
188  //We should have removed all the nodes from the free list
189  BOOST_ASSERT(m_freelist.empty());
190 
191  //Now pass all the node to the free list again
192  m_freelist.splice_after
193  ( m_freelist.before_begin()
194  , backup_list
195  , backup_list.before_begin()
196  , backup_list_last
197  , backup_list.size());
198  }

205  void purge_blocks()
206  {
207  //check for memory leaks
208  BOOST_ASSERT(m_allocated==0);
209  size_type blocksize = get_rounded_size
210  (m_real_node_size*m_nodes_per_block, (size_type)alignment_of<node_t>::value);
211  typename blockslist_t::iterator
212  it(m_blocklist.begin()), itend(m_blocklist.end()), aux;
213 
214  //We iterate though the NodeBlock list to free the memory
215  while(!m_blocklist.empty()){
216  void *addr = get_block_from_hook(&m_blocklist.front(), blocksize);
217  m_blocklist.pop_front();
218  mp_segment_mngr_base->deallocate((void*)addr);
219  }
220  //Just clear free node list
221  m_freelist.clear();
222  }

274  node_t *priv_alloc_node()
275  {
276  //If there are no free nodes we allocate a new block
277  if (m_freelist.empty())
278  this->priv_alloc_block(1);
279  //We take the first free node
280  node_t *n = (node_t*)&m_freelist.front();
281  m_freelist.pop_front();
282  ++m_allocated;
283  return n;

288  void priv_dealloc_node(void *pElem)
289  {
290  //We put the node at the beginning of the free node list
291  node_t * to_deallocate = static_cast<node_t*>(pElem);

298  void priv_alloc_block(size_type num_blocks)
299  {
300  BOOST_ASSERT(num_blocks > 0);
301  size_type blocksize =
302  get_rounded_size(m_real_node_size*m_nodes_per_block, (size_type)alignment_of<node_t>::value);
303 
304  BOOST_TRY{
305  for(size_type i = 0; i != num_blocks; ++i){
306  //We allocate a new NodeBlock and put it as first
307  //element in the free Node list
308  char *pNode = reinterpret_cast<char*>
309  (mp_segment_mngr_base->allocate(blocksize + sizeof(node_t)));
310  char *pBlock = pNode;
311  m_blocklist.push_front(get_block_hook(pBlock, blocksize));
312 
313  //We initialize all Nodes in Node Block to insert
314  //them in the free Node list
315  for(size_type j = 0; j < m_nodes_per_block; ++j, pNode += m_real_node_size){
316  m_freelist.push_front(*new (pNode) node_t);
317  }
318  }
319  }
320  BOOST_CATCH(...){
321  //to-do: if possible, an efficient way to deallocate allocated blocks
322  BOOST_RETHROW
323  }
324  BOOST_CATCH_END
325  }
